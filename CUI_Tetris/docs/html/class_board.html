블럭이 끝에 도달했을 때, 이를 보드에 기록한다. 줄이 꽉찼다면 보드를 새로 고친다.

tetris_board.h 파일의 124 번째 라인에서 정의되었습니다.

125  {
126  // 현재 블럭의 정보를 보드에 반영한다.
127  Block::Color currentBlockColor = handler.blk->getColor();
128  for (auto & blkInfo : handler.blk->getBlockInfo())
129  {
130  // 좌표와 배열 subscript 사이에는 반대 관계가 존재함에 유의
131  board[handler.currPos.y + blkInfo.y][handler.currPos.x + blkInfo.x].occupied = true;
132  board[handler.currPos.y + blkInfo.y][handler.currPos.x + blkInfo.x].color = currentBlockColor;
133  }
134 
135  // 꽉찬 줄이 있는지 검사한다.
136  std::vector<bool> lineChecker;
137  int count;
138  bool refreshFlag = false;
139  // 보드의 위에서부터 아래로 꽉찬 줄을 검사한다.
140  for (int i = height - 1; i >= 0; --i)
141  {
142  count = 0;
143  // 이번 줄의 찬 자리를 센다.
144  for (int j = 0; j < width; ++j)
145  {
146  if (board[i][j].occupied == true)
147  ++count;
148  else
149  break;
150  }
151 
152  // 줄이 꽉찼다면 false, 반대라면 true를 push한다.
153  if (count == width)
154  {
155  lineChecker.push_back(false);
156  refreshFlag = true;
157  }
158  else
159  lineChecker.push_back(true);
160  }
161 
162  // 만약 꽉찬 줄이 하나라도 있다면, 보드를 새로 고친다.
163  if (refreshFlag == true)
164  {
165  // 새 보드를 생성하여 기존 보드로부터 정보를 받되, 꽉 찬 줄은 스킵한다.
166  BoardType newBoard {};
167  int j = lineChecker.size() - 1;
168  for (int i = lineChecker.size() - 1; i >= 0; --i)
169  if (lineChecker[lineChecker.size() - 1 - i] == true)
170  {
171  for (int k = 0; k < width; ++k)
172  newBoard[j][k] = board[i][k];
173  --j;
174  }
175  board = newBoard;
176  }
177 }
